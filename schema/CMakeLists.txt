INCLUDE(InstallFilesRecursive)

FOREACH(FORMAT ${FORMATS})
    FOREACH(DOMAIN ${DOMAINS})
        STRING(TOLOWER "${FORMAT}" FORMAT_LOWER)
        STRING(TOLOWER "${DOMAIN}" DOMAIN_LOWER)
        INSTALL_FILES_RECURSIVE("share/rst/schema/" ${FORMAT}_${DOMAIN}_FILES)
    ENDFOREACH()
ENDFOREACH()

FOREACH(TARGET ${MAPPING_TARGETS})
    FOREACH(DOMAIN ${DOMAINS})
        INSTALL_FILES_RECURSIVE("share/rst/schema/" MAPPING_${TARGET}_${DOMAIN}_FILES)
    ENDFOREACH()
ENDFOREACH()
